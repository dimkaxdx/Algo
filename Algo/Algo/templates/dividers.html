Факторизация числа

Возможно, не очень понятно, зачем нужна проверка 12й строчке – если i является корнем числа x, то обратного делителя не существует.

Работает за O( sqrt(N) ):

  1. #include <bits/stdc++.h>
  2.  
  3. using namespace std;
  4.  
  5. vector<int> getDividers(int x)
  6. {
  7.     vector<int> dividers;
  8.     for (int i = 1 ; i <= sqrt(x) ; i ++)
  9.         if (x % i == 0)
  10.         {
  11.             dividers.push_back(i);
  12.             if (i * i != x)
  13.                 dividers.push_back(x/i);
  14.         }
  15.     return dividers;
  16. }
  17.  
  18. int main()
  19. {
  20.     int num; cin >> num;
  21.     vector<int> dividers = getDividers(num);
  22.     sort(dividers.begin(), dividers.end()); //сортировка для красоты вывода
  23.     cout << "Dividers for " << num << ": ";
  24.     for(auto div : dividers)
  25.         cout << div << ' ';
  26. }
