Системы счисления

Перевод в любую систему счисления, и обратное преобразование.

mypow используется из-за кривой реализации pow в С++.

  1. #include <bits/stdc++.h>
  2.  
  3. using namespace std;
  4.  
  5. int mypow(int a, int n)
  6. {
  7. 	int res = 1;
  8. 	while (n)
  9.     {
  10. 		if (n & 1)
  11. 			res *= a;
  12. 		a *= a;
  13. 		n >>= 1;
  14. 	}
  15. 	return res;
  16. }
  17.  
  18. int char_to_int(char a)
  19. {
  20.     if('0' <= a && a <= '9')
  21.         return a-'0';
  22.     return (a-'A') + 10;
  23. }
  24. int int_to_char(int a)
  25. {
  26.     if(a < 10)
  27.         return a+'0';
  28.     return (a - 10)+'A';
  29. }
  30.  
  31. string convert(int num, int sys)
  32. {
  33.     string ans = "";
  34.     while(num)
  35.     {
  36.         ans += int_to_char(num % sys);
  37.         num /= sys;
  38.     }
  39.     reverse(ans.begin(), ans.end());
  40.     return ans;
  41. }
  42. int deconvert(string num, int sys)
  43. {
  44.     int ans = 0;
  45.     for(int i = 0 ; i < (int)num.size() ; i ++)
  46.         ans += char_to_int(num[i]) * mypow(sys, num.size()-i-1);
  47.     return ans;
  48. }
  49.  
  50. int main()
  51. {
  52.     cout << "Введите число и систему\n";
  53.     int num, sys; cin >> num >> sys;
  54.     string ans = convert(num, sys);
  55.     cout << ans << endl;
  56.     cout << deconvert(ans, sys) << endl;
  57. }
