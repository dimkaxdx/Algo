BFS – обход в ширину.

BFS – один из основных алгоритмов на графах.

Суть заключается в параллельном обходе всех вершин компоненты связаности откуда мы начинали обход.

Иногда данный обход называют "поджогом", из-за того, что обход графа распространяется паралелльно по всем направлениям.

Очевидно, что из-за такого метода обхода верно утверждение, что если мы на текущей итерации находимя в вершине i и пытаемся перейти в врешину j, то dst[j] = dst[i] + 1.

  1. #include <bits/stdc++.h>
  2.  
  3. using namespace std;
  4.  
  5. const int N = 1e5 + 7;
  6.  
  7. vector<int> g[N];
  8. bool used[N];
  9. int dst[N];
  10.  
  11. int main()
  12. {
  13.     memset(dst, INT_MAX, sizeof(dst));
  14.     int n, m; cin >> n >> m;
  15.     for(int i = 0 ; i < m ; i ++)
  16.     {
  17.         int u, v; cin >> u >> v;
  18.         --u, --v;
  19.         g[u].push_back(v);
  20.         g[v].push_back(u);
  21.     }
  22.     queue<int> q;
  23.     q.push(0); used[0] = true; dst[0] = 0;
  24.     while(!q.empty())
  25.     {
  26.         int cur = q.front(); q.pop();
  27.         for(int next : g[cur])
  28.         {
  29.             if(used[next])
  30.                 continue;
  31.             q.push(next);
  32.             used[next] = true;
  33.             dst[next] = dst[cur] + 1;
  34.         }
  35.     }
  36.     for(int i = 0 ; i < n ; i ++)
  37.         if(dst[i] == INT_MAX)
  38.             cout << "Вершина " << i+1 << " недостежима из вершины 1.\n";
  39.         else
  40.             cout << "Дистанция от вершины 1 до " << i+1 << " равна " << dst[i] << ".\n";
  41. }

Данный алгоритм очень легко модифицировать, ради примера добавим восстановление пути.

  1. #include <bits/stdc++.h>
  2.  
  3. using namespace std;
  4.  
  5. const int N = 1e5 + 7;
  6.  
  7. vector<int> g[N];
  8. bool used[N];
  9. int dst[N], pr[N];
  10.  
  11. int main()
  12. {
  13.     memset(dst, INT_MAX, sizeof(dst));
  14.     memset(pr, -1, sizeof(pr));
  15.     int n, m; cin >> n >> m;
  16.     for(int i = 0 ; i < m ; i ++)
  17.     {
  18.         int u, v; cin >> u >> v;
  19.         --u, --v;
  20.         g[u].push_back(v);
  21.         g[v].push_back(u);
  22.     }
  23.     queue<int> q;
  24.     q.push(0); used[0] = true; dst[0] = 0;
  25.     while(!q.empty())
  26.     {
  27.         int cur = q.front(); q.pop();
  28.         for(int next : g[cur])
  29.         {
  30.             if(used[next])
  31.                 continue;
  32.             q.push(next);
  33.             used[next] = true;
  34.             dst[next] = dst[cur] + 1;
  35.             pr[next] = cur;
  36.         }
  37.     }
  38.  
  39.     int k;
  40.     cout << "Введите номер вершины, путь до которой вас интересует: "; cin >> k; --k;
  41.  
  42.     int cur = k; vector<int> p;
  43.     while(cur != -1)
  44.     {
  45.         p.push_back(cur);
  46.         cur = pr[cur];
  47.     }
  48.     reverse(p.begin(), p.end());
  49.  
  50.     cout << "Наикратчайший пусть из вершины 1 до " << k+1 << ":\n";
  51.     if(p.size() == 1)
  52.         cout << "Вершина " << k+1 << " недостежима из вершины 1.\n";
  53.     else
  54.         for(int c : p)
  55.             cout << c+1 << ' ';
  56. }
