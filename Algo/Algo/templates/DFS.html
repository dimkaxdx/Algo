DFS – обход в глубину.

DFS – один из основных алгоритмов на графах.

DFS – это алгоритм, в отличии от BFS, для поиска случайного пути, а не кратчайшего.

  10. void dfs(int v)
  11. {
  12.     cout << v+1 << ' ';
  13.     for(auto next : g[v])
  14.     {
  15.         if(!used[next])
  16.         {
  17.             used[next] = true;

  20.     }
  21. }
