Факторизация числа

Алгоритм слишком прост, что бы объяснять. Работает за O( sqrt(N) ):

  1. #include <bits/stdc++.h>
  2.  
  3. using namespace std;
  4.  
  5. vector<int> factorize(int x)
  6. {
  7.     vector<int> factors;
  8.     for (int i = 2 ; i <= sqrt(x) ; i ++)
  9.     {
  10.         while (x % i == 0)
  11.         {
  12.             factors.push_back(i);
  13.             x /= i;
  14.         }
  15.     }
  16.     if (x != 1)
  17.         factors.push_back(x);
  18.     return factors;
  19. }
  20.  
  21. int main()
  22. {
  23.     int num; cin >> num;
  24.     vector<int> factors = factorize(num);
  25.     cout << "Factors for " << num << ": ";
  26.     for(auto fact : factors)
  27.         cout << fact << ' ';
  28. }
